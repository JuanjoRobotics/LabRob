35/* Function ====================================================================



36 * Pause the process (w/o hogging the cpu) until the XCP Master has modified the



37 * xcpModelStartRequest value to true or until we are no longer in the paused state.  



38 * The XCP stack must continue to process events (otherwise the host would not 



39 * be able to communicate with the target).



40 */



41void rtExtModePauseIfNeeded(RTWExtModeInfo *ei,



42    int_T          numSampTimes,



43    boolean_T      *stopReqPtr)



44{



45    XCP_UNUSED_PARAM(ei);



46    XCP_UNUSED_PARAM(numSampTimes);



47



48    if (stopReqPtr == NULL) {



49        XCP_PRINTF("rtExtModePauseIfNeeded error: invalid stopReqPtr\n");



50        return;



51    }



52



53    while ((xcpModelStatus == TARGET_STATUS_PAUSED) && !xcpModelStartRequest && !(*stopReqPtr)) {



54        XCP_SLEEP(0, EXTMODE_RETRY_TIME_IN_MICROSECONDS);



55        xcpExtModeRunBackground(DEFAULT_XCP_EXTMODE_RUN_BACKGROUND_FLUSH);



56        *stopReqPtr = xcpModelStopRequest;



57    }



58



59    xcpModelStartRequest = false; /* reset to false - if we were stepped we want to



60                                   *                  stop again next time we get



61                                   *                  back here.



62                                   */



63} /* end rtExtModePauseIfNeeded */



64



65/* Function ====================================================================



66 * Pause the process (w/o hogging the cpu) until the XCP Master has modified the



67 * xcpModelStart value to true.  



68 * The XCP stack must continue to process events (otherwise the host would not 



69 * be able to communicate with the target)



70 */



71void rtExtModeWaitForStartPkt(RTWExtModeInfo *ei,



72                              int_T          numSampTimes,



73                              boolean_T      *stopReqPtr)



74{



75    XCP_UNUSED_PARAM(numSampTimes);



76



77    if (stopReqPtr == NULL) {



78        XCP_PRINTF("rtExtModeWaitForStartPkt error: invalid stopReqPtr\n");



79        return;



80    }



81



82    /* Initialize the model checksum information */



83    xcpModelChecksum0 = rteiGetChecksum0(ei);



84    xcpModelChecksum1 = rteiGetChecksum1(ei);



85    xcpModelChecksum2 = rteiGetChecksum2(ei);



86    xcpModelChecksum3 = rteiGetChecksum3(ei);



87



88#if defined(INTEGER_CODE) && INTEGER_CODE == 1



89    xcpModelIntegerCode = 1;



90#else



91    xcpModelIntegerCode = 0;



92#endif



93



94    /*



95     * Pause until the XCP master modifies the xcpModelStartRequest value



96     */



97    while(!xcpModelStartRequest && !(*stopReqPtr)) {



98        XCP_SLEEP(0L, EXTMODE_RETRY_TIME_IN_MICROSECONDS);



99        xcpExtModeRunBackground(DEFAULT_XCP_EXTMODE_RUN_BACKGROUND_FLUSH);



100        *stopReqPtr = xcpModelStopRequest;



101    }



102    if (xcpModelStatus != TARGET_STATUS_PAUSED) {



103        xcpModelStatus = TARGET_STATUS_RUNNING;



104    } else {



105        /* leave in pause mode */



106    }



107}

130void rtExtModeUpload(int_T tid, real_T taskTime)



131{



132    XcpErrorCode errorCode = XCP_SUCCESS;



133    extmodeEventId_T eventId = 0;



134    extmodeSimulationTime_T simulationTime = (extmodeSimulationTime_T)  taskTime;



135    boolean_T triggerEnable = true;



136



137    if ((tid < 0) || (tid > EXTMODE_MAX_EVENT_ID)) {



138        XCP_PRINTF("rtExtModeUpload error: invalid tid (%d)\n", tid);



139        return;



140    }



141



142    eventId = (extmodeEventId_T) tid;



143



144#ifdef EXTMODE_XCP_TRIGGER_SUPPORT



145    {



146        XcpStatus xcpStatus = xcpGetStatus();



147        if (xcpStatus == XCP_SYNC_DATA_TRANSFER) {



148            triggerEnable = xcpExtModeClassicTriggerEnabled(eventId);



149        }



150    }



151#endif



152



153    {



154#ifdef XCP_TIMESTAMP_BASED_ON_SIMULATION_TIME



155        /* Update local absolute time variables*/



156        uint32_T timestampBasedOnSimTime = xcpExtModeGetUpdatedTimestamp(simulationTime, eventId);



157#else



158        /* Update local absolute time variables*/



159        xcpExtModeUpdateTime(simulationTime, eventId);



160#endif



161



162        if (triggerEnable) {



163#ifdef XCP_TIMESTAMP_BASED_ON_SIMULATION_TIME



164            /* Notify XCP Stack about the eventId */



165            errorCode = xcpEventExternalTimestamp(eventId, timestampBasedOnSimTime);



166#else



167            /* Notify XCP Stack about the eventId, using the timestamp read



168           from the HW timer supported by XCP Platform Abstraction layer */



169            errorCode = xcpEvent(eventId);



170#endif



171        }



172    }



173



174    if (errorCode != XCP_SUCCESS) {



175        XCP_PRINTF("rtExtModeUpload error: code %d\n", errorCode);



176    }



177}

217void rtExtModeParseArgs(int_T        argc, 



218                        const char_T *argv[],



219                        real_T       *unused)



220{



221    UNUSED_PARAMETER(unused);



222



223#ifdef  EXTMODE_DISABLE_ARGS_PROCESSING



224



225    /* Some targets do not support command line args */



226    UNUSED_PARAMETER(argc);



227    UNUSED_PARAMETER(argv);



228



229#else   



230



231    {/* Extract the "-w" option (wait for packet from host), if present.



232        Note: in Classic External mode this parameter is handled within rtiostream_interface.c,



233        but this file is no longer present in XCP-based External Mode.



234        Let all the remaining parameters pass through to XCP-based external mode for parsing.  



235        NULL out the "-w" argument so that it can then be ignored once handled. */



236        int_T  count = 1;



237



238        while (count < argc) {



239            const char_T *option = argv[count++];



240



241            if ((option != NULL) && (strcmp(option, "-w") == 0)) {



242                xcpModelStartRequest = false;



243                argv[count - 1] = NULL;



244            }



245        }



246    }



247



248    /* Extract XCP specific initialization parameters */



249    xcpExtModeParseArgs(argc, argv);



250



251#endif  /* EXTMODE_DISABLE_ARGS_PROCESSING */



252}

268}

282void rtParseArgsForExtMode(int_T        argc, 



283                           const char_T *argv[])



284{



285#ifdef  EXTMODE_DISABLE_ARGS_PROCESSING



286



287    /* Some targets do not support command line args */



288    UNUSED_PARAMETER(argc);



289    UNUSED_PARAMETER(argv);



290



291#else    



292



293    /* parse command line args */



294    int_T  oldStyle_argc;



295    const char_T *oldStyle_argv[5];



296



297    if ((argc > 1) && (argv[1][0] != '-')) {



298        /* old style */



299        if ( argc > 3 ) {



300            displayUsage();



301            exit(EXIT_FAILURE);



302        }



303



304        oldStyle_argc    = 1;



305        oldStyle_argv[0] = argv[0];



306    



307        if (argc >= 2) {



308            oldStyle_argc = 3;



309



310            oldStyle_argv[1] = "-tf";



311            oldStyle_argv[2] = argv[1];



312        }



313



314        if (argc == 3) {



315            oldStyle_argc = 5;



316



317            oldStyle_argv[3] = "-port";



318            oldStyle_argv[4] = argv[2];



319        }



320



321        argc = oldStyle_argc;



322        argv = oldStyle_argv;



323    }



324



325    {



326        /* new style: */



327    	extmodeSimulationTime_T  timeValue = 0;



328        char_T tmpStr2[200];



329        int_T  count      = 1;



330        int_T  parseError = false;



331



332        /*



333         * Parse the standard RTW parameters.  Let all unrecognized parameters



334         * pass through to external mode for parsing.  NULL out all args handled



335         * so that the external mode parsing can ignore them.



336         */



337        while (count < argc) {



338            const char_T *option = argv[count++];



339



340            if (option != NULL) {



341                /* final time */



342                if ((strcmp(option, "-tf") == 0) && (count != argc)) {



343                    const char_T *tfStr = argv[count++];



344



345                    sscanf(tfStr, "%200s", tmpStr2);



346                    if (strcmp(tmpStr2, "inf") == 0) {



347                        timeValue = EXTMODE_SIMULATION_RUN_FOREVER;



348                    }



349                    else {



350                        char_T tmpstr[2];



351#if defined(INTEGER_CODE) && INTEGER_CODE == 1



352                        int tmpValue;



353



354                        if ((sscanf(tmpStr2, "%d%1s", &tmpValue, tmpstr) != 1) ||



355                            (tmpValue < 0)) {



356                            XCP_PRINTF("External mode final simulation time must be a positive, integer value or inf\n");



357                            parseError = true;



358                            break;



359                        }



360                        timeValue = (extmodeSimulationTime_T) tmpValue;



361#else



362                        if ((sscanf(tmpStr2, "%lf%1s", &timeValue, tmpstr) != 1) ||



363                            (timeValue < (real_T)0)) {



364                            XCP_PRINTF("External mode final simulation time must be a positive, real value or inf\n");



365                            parseError = true;



366                            break;



367                        }



368#endif



369                    }



370                    xcpExtmodeFinalSimulationTime = timeValue;



371



372                    argv[count - 2] = NULL;



373                    argv[count - 1] = NULL;



374                }



375                /* -w (wait for packet from host) option */



376                else if (strcmp(option, "-w") == 0) {



377                    xcpModelStartRequest = false;



378                    argv[count - 1] = NULL;



379                }



380            }



381        }



382



383        if (parseError) {



384            XCP_PRINTF("\nUsage: model_name -option1 val1 -option2 val2 -option3 "



385                       "...\n\n");



386#if defined(INTEGER_CODE) && INTEGER_CODE == 1



387            XCP_PRINTF("\t-tf 20 - sets final time to 20 ticks\n");



388#else



389            XCP_PRINTF("\t-tf 20 - sets final time to 20 seconds\n");



390#endif



391            exit(EXIT_FAILURE);



392        }



393



394        /* Extract XCP specific initialization parameters */



395        xcpExtModeParseArgs(argc, argv);



396



397        /*



398         * Check for unprocessed ("unhandled") args.



399         */



400        {



401            int i;



402            for (i=1; i<argc; i++) {



403                if (argv[i] != NULL) {



404                    XCP_PRINTF("Unexpected command line argument: %s\n", argv[i]);



405                    exit(EXIT_FAILURE);



406                }



407            }



408        }



409    }



410



411    if (xcpExtmodeFinalSimulationTime == EXTMODE_SIMULATION_RUN_FOREVER) {



412        XCP_PRINTF("\n**warning: the simulation will run with no stop time due "



413                   "to external mode with '-tf inf' argument.\n");



414    }



415



416#endif /* EXTMODE_DISABLE_ARGS_PROCESSING */



417}
