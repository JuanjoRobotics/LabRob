25static boolean_T isValidShortDownloadSize(uint8_T size)



26{



27    size_t addressGranularity = XCP_ADDRESS_GRANULARITY_BYTES_NUMBER;



28    size_t maxCtoSize = xcpTransportMaxCtoSize();



29



30    return (size <= ((maxCtoSize - 8) / addressGranularity));



31}



32



33#if XCP_SET_MTA_ENABLE == 1



34    static boolean_T isValidDownloadSize(uint8_T size)



35    {



36        size_t addressGranularity = XCP_ADDRESS_GRANULARITY_BYTES_NUMBER;



37        size_t maxCtoSize = xcpTransportMaxCtoSize();



38        



39        return ( (size >= 1)  && (size <= (maxCtoSize-2) / addressGranularity ) );



40    }



41#endif  /* XCP_SET_MTA_ENABLE == 1 */

43/*****************************************************************************



44    XCP DOWNLOAD



45******************************************************************************/



46#if XCP_SET_MTA_ENABLE == 1



47    static XcpProtoErrorCode downloadInputPacketHandler(void   *msgBuffer,



48        size_t  xcpPacketOffset,



49        size_t *outputPacketSize)



50    {



51        XcpProtoErrorCode protoErrorCode = XCP_PROTO_SUCCESS;



52        uint8_T *packet = (uint8_T *)msgBuffer + xcpPacketOffset;



53        XcpDownloadCmdPacketFrame *frame = (XcpDownloadCmdPacketFrame *)packet;



54        uint8_T *address = NULL;



55        boolean_T ok = false;



56



57        /* Check if the number of data elements is valid */



58        ok = isValidDownloadSize((uint8_T)frame->size);



59        XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE,



60            ("DOWNLOAD: invalid number of data elements (%d)\n", frame->size));



61



62        /* Retrieve memory address */



63        address = xcpStandardGetAddressFromMta();



64



65        XCP_PRINTF("DOWNLOAD: writing at address %p\n", address);



66



67        /* Update the value of the parameter with the data element content */



68        XCP_MEMCPY(address, packet + sizeof(XcpDownloadCmdPacketFrame), (uint8_T)frame->size);



69



70        /* Post-increment MTA by the frame size in address granularity units */



71        xcpStandardIncrementMta((uint8_T)frame->size);



72



73        *outputPacketSize = XCP_GENERIC_RES_PACKET_SIZE_IN_BYTES;



74



75        return protoErrorCode;



76    }



77#endif  /* XCP_SET_MTA_ENABLE == 1 */

79/*****************************************************************************



80    XCP SHORT_DOWNLOAD



81******************************************************************************/



82static XcpProtoErrorCode shortDownloadInputPacketHandler(void   *msgBuffer,



83    size_t  xcpPacketOffset,



84    size_t *outputPacketSize)



85{



86    XcpProtoErrorCode protoErrorCode = XCP_PROTO_SUCCESS;



87    uint8_T *packet = (uint8_T *)msgBuffer + xcpPacketOffset;



88    XcpShortDownloadCmdPacketFrame *frame = (XcpShortDownloadCmdPacketFrame *)packet;



89    uint8_T *address = NULL;



90    boolean_T ok = false;



91



92    /* Check if the number of data elements is valid */



93    ok = isValidShortDownloadSize((uint8_T)frame->size);



94    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE,



95        ("SHORT_DOWNLOAD: invalid number of data elements (%d)\n", frame->size));



96



97    /* Check if the memory address is valid and set the MTA pointer */



98    ok = xcpStandardSetMta(frame->address, (uint8_T)frame->addressExtension);



99    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_ACCESS_DENIED, ("SHORT_DOWNLOAD: invalid address\n"));



100



101    /* Retrieve memory address */



102    address = xcpStandardGetAddressFromMta();



103



104    XCP_PRINTF("SHORT_DOWNLOAD: writing at address %p\n", address);



105



106    /* Update the value of the parameter with the data element content */



107    XCP_MEMCPY(address, packet + sizeof(XcpShortDownloadCmdPacketFrame), (uint8_T)frame->size);



108



109    /* Post-increment MTA by the frame size in address granularity units */



110    xcpStandardIncrementMta((uint8_T)frame->size);



111



112    *outputPacketSize = XCP_GENERIC_RES_PACKET_SIZE_IN_BYTES;



113



114    return protoErrorCode;



115}

119static XcpPacketHandlers calibrationSupportedRxPacket[] =

121#if XCP_SET_MTA_ENABLE == 1

128/*****************************************************************************



129    XCP Packet Lookup Function for basic calibration commands

132/* Default Calibration Packet Lookup function, supporting only basic commands



133   listed in the table above */



134static XcpPacketHandlers* getPacket(XcpRxPidCode pid)



135{



136    return xcpFindPacket(pid, calibrationSupportedRxPacket,



137                         XCP_ELEMENTS_NUMBER(calibrationSupportedRxPacket));



138}
