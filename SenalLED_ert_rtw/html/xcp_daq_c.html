52static boolean_T isActiveDaqList(uint16_T daqListId)



53{



54    return (xcpDynamicDaqLists.daq[daqListId].status == XCP_DAQ_SELECTED) ||



55           (xcpDynamicDaqLists.daq[daqListId].status == XCP_DAQ_STARTED);



56}

58static boolean_T noActiveDaqLists(void)



59{



60    boolean_T found = 0;



61    uint16_T i = 0;



62



63    for (i = 0; (i < xcpDynamicDaqLists.daqCount) && !found; i++) {



64        found = isActiveDaqList(i);



65    }



66



67    return !found;



68}

70static boolean_T isValidDaqPtr(uint16_T daqListId, uint8_T odtId, uint8_T odtEntryId)



71{



72    boolean_T ret = (xcpDynamicDaqLists.daq != NULL) &&



73                    (daqListId < xcpDynamicDaqLists.daqCount) &&



74                    (xcpDynamicDaqLists.daq[daqListId].odt != NULL) &&



75                    (odtId < xcpDynamicDaqLists.daq[daqListId].odtCount) &&



76                    (xcpDynamicDaqLists.daq[daqListId].odt[odtId].entry != NULL) &&



77                    (odtEntryId < xcpDynamicDaqLists.daq[daqListId].odt[odtId].entriesCount);



78



79    return ret;



80}

82static boolean_T isValidDaqEntry(uint8_T bitOffset, uint8_T size, uint32_T address)



83{



84    boolean_T ret = false;



85    uint16_T  entrySize = size;     /* cast to 16-bit avoids -Werror=type-limits */



86



87    if (bitOffset == 0xFF) {



88        /* "Normal" (non-bitwise) access has been selected */



89        if ( (entrySize <= XCP_MAX_ODT_ENTRY_SIZE) &&



90             (entrySize % (XCP_ODT_ENTRY_SIZE_GRANULARITY/XCP_ADDRESS_GRANULARITY_BYTES_NUMBER) == 0) &&



91             (  address % (XCP_ODT_ENTRY_SIZE_GRANULARITY/XCP_ADDRESS_GRANULARITY_BYTES_NUMBER) == 0) ) {



92            ret = true;



93        }



94    }



95    else {



96        /* Bitwise access has been selected */



97        if ((bitOffset <= 0x1F) && (address % XCP_ODT_ENTRY_SIZE_GRANULARITY) &&



98            (entrySize == XCP_ODT_ENTRY_SIZE_GRANULARITY)) {



99            ret = true;



100        }



101    }



102



103    return ret;



104}

106/** The current supported mode includes:



107- alternating = 0



108- direction = 0 (DAQ only)



109- timestamp = 1 or 0



110- PID OFF = 0



111*/



112static boolean_T isSupportedDaqListMode(uint8_T mode)



113{



114    #if XCP_TIMESTAMP_FIXED == 0



115        return ( (mode & ~XCP_DAQ_MODE_TIMESTAMP_MASK) == 0 );



116    #else



117        return (mode == XCP_DAQ_MODE_TIMESTAMP_MASK);



118    #endif



119}

121static boolean_T isValidStartStopMode(uint8_T mode)



122{



123    boolean_T ret = false;



124



125    if ((mode == XCP_DAQ_LIST_STOP) ||



126        (mode == XCP_DAQ_LIST_START) ||



127        (mode == XCP_DAQ_LIST_SELECT))



128        ret = true;



129



130    return ret;



131}

134{

138        (mode == XCP_DAQ_LIST_START_SELECTED) ||



139        (mode == XCP_DAQ_LIST_STOP_SELECTED))



140        ret = true;



141



142    return ret;

150/* Try to allocate a reserved memory pool dedicated to the packets in the DAQ list.



151   @note depending on the configuration of the memory allocator, this may fail



152         so we just set daq->poolId to XCP_INVALID_POOL_ID in that case */



153static XcpErrorCode createDaqReservedPool(XcpDaq *daq)



154{



155    XcpErrorCode errorCode = XCP_SUCCESS;



156    size_t poolBlockSize = 0;



157    int i = 0;



158



159    /* Calculate the max size for the ODT packets in the given DAQ list */



160    for (i = 0; i < daq->odtCount; i++) {



161        /* Calculate the required message buffer size */



162        size_t packetSize = XCP_IN_BYTES(daq->odt[i].size + XCP_IN_AG(XCP_ID_FIELD_SIZE));



163        size_t msgBufferSize = 0;



164



165        if (i == 0) {



166            /* The first DTO should also contain the Timestamp info */



167            packetSize += XCP_TIMESTAMP_SIZE;



168        }



169



170        msgBufferSize = xcpTransportMsgBufferSize(packetSize, XCP_DTO);



171



172        if (msgBufferSize > poolBlockSize) {



173            poolBlockSize = msgBufferSize;



174        }



175    }



176



177    /* Try the allocation of the dedicate pool */



178    errorCode = xcpMemReservedPoolCreate(poolBlockSize,



179    		                             XCP_MEM_DAQ_RESERVED_POOL_BLOCKS_NUMBER,



180                                         &daq->poolId);



181    if (errorCode != XCP_SUCCESS) {



182        daq->poolId = XCP_INVALID_POOL_ID;



183    }



184



185    return errorCode;



186}



187



188static void destroyDaqReservedPool(XcpDaq *daq)



189{



190    if (daq->poolId != XCP_INVALID_POOL_ID) {



191        xcpMemReservedPoolDestroy(daq->poolId);



192    }



193



194    daq->poolId = XCP_INVALID_POOL_ID;



195}

198/*****************************************************************************



199    XCP SET DAQ PTR



200******************************************************************************/



201static XcpProtoErrorCode setDaqPtrInputPacketHandler(void   *msgBuffer,



202                                                     size_t  xcpPacketOffset,



203                                                     size_t *outputPacketSize)



204{



205    XcpProtoErrorCode protoErrorCode = XCP_PROTO_SUCCESS;



206    uint8_T *packet = (uint8_T *)msgBuffer + xcpPacketOffset;



207    XcpSetDaqPtrCmdPacketFrame *frame = (XcpSetDaqPtrCmdPacketFrame *) packet;



208    uint16_T daqListId  = (uint16_T)frame->daqListId;



209    uint8_T  odtId      = (uint8_T)frame->odtId;



210    uint8_T  odtEntryId = (uint8_T)frame->odtEntryId;



211    boolean_T ok = false;



212



213    /* Check if the selected ODT Entry is available */



214    ok = isValidDaqPtr(daqListId, odtId, odtEntryId);



215    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE,



216        ("SET_DAQ_PTR: ODT entry not available (DAQ %d, ODT %d, Entry %d)\n", daqListId, odtId, odtEntryId));



217



218    /* Check if the DAQ list is currently active */



219    ok = !isActiveDaqList(daqListId);



220    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_DAQ_ACTIVE, ("SET_DAQ_PTR: DAQ list %d is currently active\n", daqListId));



221



222    /* Everything is fine -> proceed and update the current ODT entry pointer */



223    XCP_PRINTF("SET_DAQ_PTR: updating current ODT entry pointer to DAQ %d, ODT %d, Entry %d\n",



224               daqListId, odtId, odtEntryId);



225



226    xcpCurrentDaq.daqListId  = daqListId;



227    xcpCurrentDaq.odtId      = odtId;



228    xcpCurrentDaq.odtEntryId = odtEntryId;



229



230    *outputPacketSize = XCP_GENERIC_RES_PACKET_SIZE_IN_BYTES;



231



232    return protoErrorCode;



233}

235/*****************************************************************************



236    XCP WRITE DAQ



237******************************************************************************/



238static XcpProtoErrorCode writeDaqInputPacketHandler(void   *msgBuffer,



239                                                    size_t  xcpPacketOffset,



240                                                    size_t *outputPacketSize)



241{



242    XcpProtoErrorCode protoErrorCode = XCP_PROTO_SUCCESS;



243    uint8_T *packet = (uint8_T *)msgBuffer + xcpPacketOffset;



244    XcpWriteDaqCmdPacketFrame *frame = (XcpWriteDaqCmdPacketFrame *)packet;



245    uint16_T daqListId = xcpCurrentDaq.daqListId;



246    uint8_T  odtId = xcpCurrentDaq.odtId;



247    uint8_T  odtEntryId = xcpCurrentDaq.odtEntryId;



248    XcpOdtEntry *entry = NULL;



249    boolean_T ok = false;



250



251    /* Check if the selected ODT Entry is available */



252    ok = isValidDaqPtr(daqListId, odtId, odtEntryId);



253    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_WRITE_PROTECTED,



254        ("WRITE_DAQ: current ODT entry not available for write operations (DAQ %d, ODT %d, Entry %d)\n", daqListId, odtId, odtEntryId));



255



256    /* Check if the DAQ list is currently active */



257    ok = !isActiveDaqList(daqListId);



258    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_DAQ_ACTIVE, ("XCP_WRITE_DAQ: DAQ list %d is currently active\n", daqListId));



259



260    /* Check if the Entry values are valid */



261    ok = isValidDaqEntry((uint8_T)frame->bitOffset, (uint8_T)frame->size, (uint32_T)frame->address);



262    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE,



263        ("XCP_WRITE_DAQ: invalid DAQ %d ODT %d Entry %d value: bitOffset = %xH size = %d addressExtension = %dH address = %xH\n",



264         daqListId, odtId, odtEntryId, frame->bitOffset, frame->size, frame->addressExtension, frame->address));



265



266    /* Everything is fine -> proceed and write the entry */



267    XCP_PRINTF("XCP_WRITE_DAQ: writing DAQ %d ODT %d Entry %d: bitOffset = %xH size = %d addressExtension = %dH address = %xH\n",



268               daqListId, odtId, odtEntryId, frame->bitOffset, frame->size, frame->addressExtension, frame->address);



269



270    entry = &(xcpDynamicDaqLists.daq[daqListId].odt[odtId].entry[odtEntryId]);



271



272    entry->bitOffset        = (uint8_T)frame->bitOffset;



273    entry->size             = (uint8_T)frame->size;



274    entry->addressExtension = (uint8_T)frame->addressExtension;



275    entry->address          = (uint32_T)frame->address;



276



277    /* After a successful WRITE_DAQ, the xcpDaqPtr odtEntryId needs to be post incremented */



278    xcpCurrentDaq.odtEntryId++;



279



280    *outputPacketSize = XCP_GENERIC_RES_PACKET_SIZE_IN_BYTES;



281



282    return protoErrorCode;



283}

285/*****************************************************************************



286    XCP SET DAQ LIST MODE



287******************************************************************************/



288static XcpProtoErrorCode setDaqListModeInputPacketHandler(void   *msgBuffer,



289                                                          size_t  xcpPacketOffset,



290                                                          size_t *outputPacketSize)



291{



292    XcpProtoErrorCode protoErrorCode = XCP_PROTO_SUCCESS;



293    uint8_T *packet = (uint8_T *)msgBuffer + xcpPacketOffset;



294    XcpSetDaqListModeCmdPacketFrame *frame = (XcpSetDaqListModeCmdPacketFrame *)packet;



295    XcpDaq *daqList = NULL;



296    boolean_T ok = false;



297



298    /* Check if the DAQ List is available */



299    ok = (xcpDynamicDaqLists.daq != NULL) && (frame->daqListId < xcpDynamicDaqLists.daqCount);



300    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE, ("SET_DAQ_LIST_MODE: DAQ list %d not available\n", frame->daqListId));



301



302    daqList = &(xcpDynamicDaqLists.daq[frame->daqListId]);



303



304    /* Check if the DAQ list is currently active */



305    ok = !isActiveDaqList((uint16_T)frame->daqListId);



306    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_DAQ_ACTIVE, ("SET_DAQ_LIST_MODE: DAQ list %d is currently active\n", frame->daqListId));



307



308    /* Check if the Event Id is valid */



309    ok = (frame->eventId < XCP_MAX_EVENT_CHANNEL);



310    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE, ("SET_DAQ_LIST_MODE: invalid event channel Id %d\n", frame->eventId));



311



312    /* Check DAQ list priority */



313    ok = (frame->priority == 0);



314    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE, ("SET_DAQ_LIST_MODE: priority not supported, must be zero\n"));



315



316    /* Check if DAQ list mode is supported */



317    ok = isSupportedDaqListMode((uint8_T)frame->mode);



318    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_MODE_NOT_VALID, ("SET_DAQ_LIST_MODE: DAQ list mode not supported %xH\n", frame->mode));



319



320    /* Everything is fine -> proceed and set the DAQ list mode */



321    XCP_PRINTF("SET_DAQ_LIST_MODE: setting DAQ list %d: mode = %xH, eventId = %d, prescaler = %d, priority = %d\n",



322        frame->daqListId, frame->mode, frame->eventId, frame->prescaler, frame->priority);



323



324    daqList->mode       = (uint8_T)frame->mode;



325    daqList->eventId    = (uint16_T)frame->eventId;



326    daqList->prescaler  = (uint8_T)frame->prescaler;



327    daqList->priority   = (uint8_T)frame->priority;



328



329    /* Assuming that prescaler = 0 means value that you are not interested in using it */



330    if (daqList->prescaler <= 1)



331        daqList->prescaler = 1;



332



333    *outputPacketSize = XCP_GENERIC_RES_PACKET_SIZE_IN_BYTES;



334



335    return protoErrorCode;



336}

338/*****************************************************************************



339    XCP START STOP DAQ LIST



340******************************************************************************/



341static XcpProtoErrorCode startStopDaqListInputPacketHandler(void   *msgBuffer,



342                                                            size_t  xcpPacketOffset,



343                                                            size_t *outputPacketSize)



344{



345    XcpProtoErrorCode protoErrorCode = XCP_PROTO_SUCCESS;



346    uint8_T *packet = (uint8_T *)msgBuffer + xcpPacketOffset;



347    XcpStartStopDaqListCmdPacketFrame *frame = (XcpStartStopDaqListCmdPacketFrame *)packet;



348    XcpDaq *daqList = NULL;



349    int32_T daqListId = frame->daqListId;



350    boolean_T ok = false;



351    boolean_T locked = false;



352



353    /* Check if the DAQ List is available */



354    ok = (xcpDynamicDaqLists.daq != NULL) && (daqListId < xcpDynamicDaqLists.daqCount);



355    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE,



356        ("START_STOP_DAQ_LIST: DAQ list %d not available\n", daqListId));



357



358    daqList = &(xcpDynamicDaqLists.daq[daqListId]);



359



360    /* Check if requested mode is valid */



361    ok = isValidStartStopMode((uint8_T)frame->mode);



362    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_MODE_NOT_VALID,



363        ("START_STOP_DAQ_LIST: invalid mode %d selected \n", frame->mode));



364



365    /* Assign the Absolute ODT number */



366    if (absoluteOdtNumberIdentification()) {



367        if (daqList->status == XCP_DAQ_INIT) {



368            /* The DAQ list has never been started/stopped/selected before



369            -> PIDs for the ODTs needs to be reserved */



370            if (XCP_READ_BIT_VALUE(daqList->mode, XCP_DAQ_MODE_DIRECTION_MASK)) {



371                /* STIM */



372                ok = (daqList->odt != NULL) &&



373                       ((xcpDynamicDaqLists.firstAvailableStimPid +



374                        daqList->odtCount) < XCP_MAX_STIM_ODT_NUMBER);



375



376                if (ok) {



377                    daqList->firstPid = xcpDynamicDaqLists.firstAvailableStimPid;



378                    xcpDynamicDaqLists.firstAvailableStimPid += daqList->odtCount;



379                }



380            }



381            else {



382                /* DAQ */



383                ok = (daqList->odt != NULL) &&



384                       ((xcpDynamicDaqLists.firstAvailableDaqPid +



385                        daqList->odtCount) < XCP_MAX_DAQ_ODT_NUMBER);



386



387                if (ok) {



388                    daqList->firstPid = xcpDynamicDaqLists.firstAvailableDaqPid;



389                    xcpDynamicDaqLists.firstAvailableDaqPid += daqList->odtCount;



390                }



391            }



392



393            XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_DAQ_CONFIG_ERROR,



394                ("START_STOP_DAQ_LIST: cannot assign PIDs for DAQ list %d \n", daqListId));



395        }



396    }



397



398    /* Check the ODT configuration and also if the ODT sizes fits the Max DTO size */



399    if ((frame->mode == XCP_DAQ_LIST_START) || (frame->mode == XCP_DAQ_LIST_SELECT)) {



400        int i = 0;



401



402        ok = true;



403        for (i = 0; (i < daqList->odtCount) && ok; i++) {



404            int j = 0;



405            size_t odtSize = 0;



406



407            /* Check if we have ODT entries */



408            ok = (daqList->odt[i].entry != NULL) && (daqList->odt[i].entriesCount > 0);



409            XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_DAQ_CONFIG_ERROR,



410                ("START_STOP_DAQ_LIST: wrong ODT entry configuration detected for DAQ list %d, ODT %d \n", daqListId, i));



411



412            /* Calculate the resulting ODT size */



413            for (j = 0; j < daqList->odt[i].entriesCount; j++) {



414                odtSize += daqList->odt[i].entry[j].size;



415            }



416



417            /* Check that the resulting ODT size is valid */



418            ok = (XCP_IN_BYTES(odtSize + XCP_IN_AG(XCP_ID_FIELD_SIZE)) + XCP_TIMESTAMP_SIZE) <= xcpTransportMaxDtoSize();



419            XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_DAQ_CONFIG_ERROR,



420                ("START_STOP_DAQ_LIST: in DAQ list %d the ODT entries size is larger than Max DTO size for ODT %d \n", daqListId, i));



421



422            /* Update the ODT size */



423            daqList->odt[i].size = odtSize;



424        }



425    }



426



427    /* Everything is fine -> proceed and try to change the DAQ list status */



428    XCP_WRITE_TRY_LOCK(xcpDaqLock, xcpRunningEventCounter, locked);



429



430    if (locked) {



431    	XcpErrorCode createPoolError = XCP_SUCCESS;



432



433        switch (frame->mode) {



434        case XCP_DAQ_LIST_STOP:



435            XCP_PRINTF("START_STOP_DAQ_LIST: stopping DAQ list %d\n", daqListId);



436            daqList->status = XCP_DAQ_STOPPED;



437            if (noActiveDaqLists()) {



438                /* update status if this was the only active DAQ list */



439                xcpStatusSet(XCP_CONNECTED);



440                xcpSessionStatusClearMask(XCP_SESSION_DAQ_RUNNING_MASK);



441            }



442            break;



443        case XCP_DAQ_LIST_START:



444            XCP_PRINTF("START_STOP_DAQ_LIST: starting DAQ list %d\n", daqListId);



445



446            /* daqList->poolId is initialized with XCP_INVALID_POOL_ID in 



447               allocDaqInputPacketHandler.



448



449               Allocate a reserved memory pool for the DAQ list, if needed. */



450            if ((daqListId < XCP_MEM_DAQ_RESERVED_POOLS_NUMBER) && 



451                (daqList->poolId == XCP_INVALID_POOL_ID)) {



452                /* Create a reserved memory pool, if not already created */



453                createPoolError = createDaqReservedPool(daqList);



454            }



455



456            if (createPoolError == XCP_SUCCESS) {



457                daqList->status = XCP_DAQ_STARTED;



458                xcpStatusSet(XCP_SYNC_DATA_TRANSFER);



459                xcpSessionStatusSetMask(XCP_SESSION_DAQ_RUNNING_MASK);



460            }



461            break;



462        case XCP_DAQ_LIST_SELECT:



463            XCP_PRINTF("START_STOP_DAQ_LIST: selected DAQ list %d\n", frame->daqListId);



464            daqList->status = XCP_DAQ_SELECTED;



465            break;



466        }



467



468        XCP_WRITE_UNLOCK(xcpDaqLock);



469



470        *outputPacketSize = XCP_START_STOP_DAQ_LIST_RES_PACKET_SIZE_IN_BYTES;



471        XCP_INPUT_PKT_ERROR_IF(createPoolError != XCP_SUCCESS, XCP_PROTO_MEMORY_OVERFLOW,



472                ("START_STOP_DAQ_LIST: cannot allocate reserved memory pool for the DAQ list\n"));



473        



474    }



475



476    /* Error if the command cannot be executed as the processing of



477       at least one event was in progress */



478    XCP_INPUT_PKT_ERROR_IF(!locked, XCP_PROTO_BUSY,



479        ("START_STOP_DAQ_LIST: cannot be carried out as an event was processed\n"));



480



481    return protoErrorCode;



482}

484static void startStopDaqListOutputPacketHandler(XcpProtoErrorCode inputCode, void *packet, size_t packetSize)



485{



486    if (inputCode == XCP_PROTO_SUCCESS) {



487        XcpStartStopDaqListCmdPacketFrame *inputFrame = (XcpStartStopDaqListCmdPacketFrame *)packet;

494    }



495    else {



496        genericOutputPacketHandler(inputCode, packet, packetSize);



497    }



498}

500/*****************************************************************************



501    XCP START STOP SYNCH



502******************************************************************************/



503static XcpProtoErrorCode startStopSynchInputPacketHandler(void   *msgBuffer,



504                                                          size_t  xcpPacketOffset,



505                                                          size_t *outputPacketSize)



506{



507    XcpProtoErrorCode protoErrorCode = XCP_PROTO_SUCCESS;



508    uint8_T *packet = (uint8_T *)msgBuffer + xcpPacketOffset;



509    XcpStartStopSynchCmdPacketFrame *frame = (XcpStartStopSynchCmdPacketFrame *)packet;



510    boolean_T ok = false;



511    boolean_T locked = false;



512    int32_T i = 0;



513



514    /* Check if requested mode is valid */



515    ok = isValidStartStopSynchMode((uint8_T)frame->mode);



516    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_MODE_NOT_VALID,



517        ("START_STOP_SYNCH: invalid mode %d selected \n", frame->mode));



518



519    /* Check if the DAQ Lists are available */



520    ok = (xcpDynamicDaqLists.daq != NULL) && (xcpDynamicDaqLists.daqCount > 0);



521    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE,



522        ("START_STOP_SYNCH: DAQ lists not available\n"));



523



524    /* Everything is fine -> proceed and try to change the DAQ list status */



525



526    /* Need to get the lock to guarantee that they are all done in the same cycle */



527    XCP_WRITE_TRY_LOCK(xcpDaqLock, xcpRunningEventCounter, locked);



528



529    if (locked) {



530        XcpErrorCode createPoolError = XCP_SUCCESS;



531        uint16_T daqListId = 0;



532



533        switch (frame->mode) {



534        case XCP_DAQ_LIST_STOP_ALL:



535            for (i = 0; i < xcpDynamicDaqLists.daqCount; i++) {



536                if (xcpDynamicDaqLists.daq[i].status != XCP_DAQ_INIT) {



537                    XCP_PRINTF("START_STOP_SYNCH: stopping DAQ list %d\n", i);



538                    xcpDynamicDaqLists.daq[i].status = XCP_DAQ_STOPPED;



539                }



540            }



541



542            /* update status as there are no more active DAQ list */



543            xcpStatusSet(XCP_CONNECTED);



544            xcpSessionStatusClearMask(XCP_SESSION_DAQ_RUNNING_MASK);



545            break;



546



547        case XCP_DAQ_LIST_START_SELECTED:



548            /* xcpDynamicDaqLists.daq[i].poolId is initialized with



549               XCP_INVALID_POOL_ID in allocDaqInputPacketHandler.



550



551               Allocate a reserved memory pool for the DAQ list, if needed. */



552            for (i = 0; (i < xcpDynamicDaqLists.daqCount) && (createPoolError == XCP_SUCCESS); i++) {



553                if (xcpDynamicDaqLists.daq[i].status == XCP_DAQ_SELECTED) {



554                    if ((i < XCP_MEM_DAQ_RESERVED_POOLS_NUMBER) && 



555                        (xcpDynamicDaqLists.daq[i].poolId == XCP_INVALID_POOL_ID)) {



556                        /* Create a reserved memory pool, if not already created */



557                        createPoolError = createDaqReservedPool(&xcpDynamicDaqLists.daq[i]);



558                        daqListId = (uint16_T) i;



559                    }



560                }



561            }



562



563            if (createPoolError == XCP_SUCCESS) {



564                /* If the allocation went well for all the DAQ lists, start them */



565                for (i = 0; (i < xcpDynamicDaqLists.daqCount); i++) {



566                    if (xcpDynamicDaqLists.daq[i].status == XCP_DAQ_SELECTED) {



567                        XCP_PRINTF("START_STOP_SYNCH: starting DAQ list %d\n", i);



568



569                        xcpDynamicDaqLists.daq[i].status = XCP_DAQ_STARTED;



570                        xcpStatusSet(XCP_SYNC_DATA_TRANSFER);



571                        xcpSessionStatusSetMask(XCP_SESSION_DAQ_RUNNING_MASK);



572                    }



573                }



574            }



575            break;



576



577        case XCP_DAQ_LIST_STOP_SELECTED:



578            for (i = 0; i < xcpDynamicDaqLists.daqCount; i++) {



579                if (xcpDynamicDaqLists.daq[i].status == XCP_DAQ_SELECTED) {



580                    XCP_PRINTF("START_STOP_SYNCH: stopping DAQ list %d\n", i);



581                    xcpDynamicDaqLists.daq[i].status = XCP_DAQ_STOPPED;



582                }



583            }



584



585            if (noActiveDaqLists()) {



586                /* update status if this was the only active DAQ list */



587                xcpStatusSet(XCP_CONNECTED);



588                xcpSessionStatusClearMask(XCP_SESSION_DAQ_RUNNING_MASK);



589            }



590            break;



591        }



592



593        XCP_WRITE_UNLOCK(xcpDaqLock);



594



595        XCP_INPUT_PKT_ERROR_IF(createPoolError != XCP_SUCCESS, XCP_PROTO_MEMORY_OVERFLOW,



596                ("START_STOP_SYNCH: cannot allocate reserved memory pool for the DAQ list %d\n", daqListId));



597



598        *outputPacketSize = XCP_GENERIC_RES_PACKET_SIZE_IN_BYTES;



599    }



600



601    /* Error if the command cannot be executed as the processing of 



602       at least one event was in progress */



603    XCP_INPUT_PKT_ERROR_IF(!locked, XCP_PROTO_BUSY,



604        ("START_STOP_SYNCH: cannot be carried out as an event was processed\n"));



605



606    return protoErrorCode;



607}

930XcpErrorCode xcpDaqEvent(XcpEventIdType eventId, uint32_T timestamp)



931{



932    XcpErrorCode errorCode = XCP_SUCCESS;



933    boolean_T error = false;



934



935    /* The lock is used to protect against the concurrent execution of XCP commands



936    (within xcpRun()) that can modify the DAQ list data structures */



937    XCP_READ_LOCK(xcpDaqLock, xcpRunningEventCounter);



938



939    if ((xcpDynamicDaqLists.daq != NULL) && (xcpDynamicDaqLists.daqCount > 0)) {



940        int i = 0;



941        XcpDaq *daq = xcpDynamicDaqLists.daq;



942



943        for (i = 0; (i < xcpDynamicDaqLists.daqCount) && !error; i++) {



944            if (!XCP_READ_BIT_VALUE(daq[i].mode, XCP_DAQ_MODE_DIRECTION_MASK) && /* it's a DAQ list */



945                (daq[i].status == XCP_DAQ_STARTED) &&                            /* it's running */



946                (daq[i].eventId == eventId)) {                                   /* and associated to this event */



947                int j = 0;



948



949                /* @note prescaler is not supported at the moment -> we send DTOs every time */



950                for (j = 0; (j < daq[i].odtCount) && !error; j++) {



951                    /* Calculate the required message buffer size */



952                    size_t packetOffset = xcpTransportPacketOffset();



953                    size_t packetSize = 0;



954                    uint8_T *msgBuffer = NULL;



955                    size_t msgBufferSize = 0;



956                    const boolean_T needsTimestamp = IS_TIMESTAMP_REQUIRED(daq[i].mode, j);



957



958                    packetSize = XCP_IN_BYTES(daq[i].odt[j].size + XCP_IN_AG(XCP_ID_FIELD_SIZE));



959                    if (needsTimestamp) {



960                        /* The first DTO should also contain the Timestamp info */



961                        packetSize += XCP_TIMESTAMP_SIZE;



962                    }



963



964                    /* Allocate the message buffer */



965                    msgBufferSize = xcpTransportMsgBufferSize(packetSize, XCP_DTO);



966



967                    if (daq[i].poolId != XCP_INVALID_POOL_ID) {



968                        /* A dedicated memory pool is available */



969                        msgBuffer = (uint8_T *)xcpMemAllocFromPool(daq[i].poolId, msgBufferSize);



970                    } else {



971                        /* No memory pool available, use the Main memory instead */



972                        msgBuffer = (uint8_T *)xcpMemAlloc(msgBufferSize);



973                    }



974



975                    if (msgBuffer == NULL) {



976                        error = true;



977                        errorCode = XCP_NO_MEMORY;



978                    }



979



980                    if (!error) {



981                        uint8_T *packet = msgBuffer + packetOffset;



982                        int k = 0;



983



984                        /* Fill the Identification Field first



985                        @note this is hard-coded to support absolute ODT number at the moment */



986                        *packet = (uint8_T)(daq[i].firstPid + j);



987                        packet++;



988



989                        /* Add the timestamp for the first DTO



990                        @note this is hard-coded to support timestamp size of 4 bytes */



991                        if (needsTimestamp) {



992                            XCP_MEMCPY(packet, &timestamp, sizeof(timestamp));



993



994                            packet += sizeof(timestamp);



995                        }



996



997                        /* Finally copy the ODT entries values */



998                        for (k = 0; k < daq[i].odt[j].entriesCount; k++) {



999                            XcpOdtEntry *entry = &(daq[i].odt[j].entry[k]);



1000                            uint8_T *address = XCP_ADDRESS_GET(entry->addressExtension, entry->address);



1001



1002                            if (daq[i].odt[j].entry[k].bitOffset != 0xFF) {



1003                                /* Bitwise access has been selected */



1004                                uint32_T *dword = (uint32_T *)address;



1005



1006                                *packet = (uint8_T) (*dword & ((uint32_T)(1 << entry->bitOffset))) != 0;



1007                            }



1008                            else {



1009                                /* 'Normal' (non-bitwise) access has been selected */



1010                                XCP_MEMCPY(packet, address, entry->size);



1011                            }



1012                            packet += entry->size;



1013                        }



1014



1015                        /* Send the response back to the Transport Layer */



1016                        errorCode = xcpTransportTxPacketSet(XCP_TRANSPORT_TX_DEFAULT_QUEUE,



1017                                                            msgBuffer, msgBufferSize,



1018                                                            packetOffset, packetSize);



1019                        if (errorCode != XCP_SUCCESS) {



1020                            xcpMemFree(msgBuffer);



1021                        }



1022                    }



1023                }



1024            }



1025        }



1026    }



1027



1028    XCP_READ_UNLOCK(xcpDaqLock, xcpRunningEventCounter);



1029



1030    return errorCode;



1031}

1034boolean_T xcpResetDaqListStatus(void)



1035{



1036    boolean_T locked = true;



1037



1038    if (xcpDynamicDaqLists.daq != NULL) {



1039        XcpDaq *daq = xcpDynamicDaqLists.daq;



1040        int32_T  daqCount = xcpDynamicDaqLists.daqCount;



1041



1042        XCP_WRITE_TRY_LOCK(xcpDaqLock, xcpRunningEventCounter, locked);



1043



1044        if (locked) {



1045            /* No events were processed, it's safe to remove the DAQ lists */



1046            int32_T i = 0;



1047            int32_T j = 0;



1048



1049            xcpDynamicDaqLists.daq = NULL;



1050            xcpDynamicDaqLists.daqCount = 0;



1051            xcpDynamicDaqLists.firstAvailableDaqPid = 0;



1052            xcpDynamicDaqLists.firstAvailableStimPid = 0;



1053



1054            XCP_WRITE_UNLOCK(xcpDaqLock);



1055



1056            /* Prepare the Transport Layer for subsequent connections 



1057              (by resetting the Frame Handler counters and deleting the 



1058              existing packets in TX/RX fifos) */



1059            xcpTransportRestart();



1060



1061            /* Delete all the Dynamic DAQ data structures and 



1062               reserved  memory pools */



1063            for (i = 0; i < daqCount; i++) {



1064                if (i < XCP_MEM_DAQ_RESERVED_POOLS_NUMBER) {



1065                    destroyDaqReservedPool(&daq[i]);



1066                }                



1067                



1068                if (daq[i].odt != NULL) {



1069                    for (j = 0; j < daq[i].odtCount; j++) {



1070                        if (daq[i].odt[j].entry != NULL) {



1071                            xcpMemFree(daq[i].odt[j].entry);



1072                        }



1073                    }



1074



1075                    xcpMemFree(daq[i].odt);



1076                }



1077            }



1078



1079            xcpMemFree(daq);



1080        }



1081



1082        xcpSessionStatusClearMask(XCP_SESSION_DAQ_RUNNING_MASK);



1083    }



1084



1085    return locked;



1086}
