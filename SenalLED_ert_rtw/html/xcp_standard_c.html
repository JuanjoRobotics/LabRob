31static boolean_T isValidUploadSize(uint8_T size)



32{



33    size_t addressGranularity = XCP_ADDRESS_GRANULARITY_BYTES_NUMBER;



34    size_t maxCtoSize = xcpTransportMaxCtoSize();



35



36    return (size < ((maxCtoSize) / addressGranularity));



37}

73static XcpProtoErrorCode connectInputPacketHandler(void   *msgBuffer,



74                                                   size_t  xcpPacketOffset,



75                                                   size_t *outputPacketSize)



76{



77    XcpProtoErrorCode protoErrorCode = XCP_PROTO_SUCCESS;



78    uint8_T *packet = (uint8_T *)msgBuffer + xcpPacketOffset;



79    XcpConnectCmdPacketFrame *frame = (XcpConnectCmdPacketFrame *) packet;



80    boolean_T ok = false;



81



82    /* If DAQ List support is enabled, reset the status of dynamic DAQ list data structures */



83    ok = xcpResetDaqListStatus();



84



85    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_BUSY,



86        ("CONNECT: DAQ lists reset cannot be carried out, as an event was processed\n"));



87



88    /* Validate command inputs */



89    ok = (frame->mode == XCP_CONNECT_MODE_NORMAL) || (frame->mode == XCP_CONNECT_MODE_USER_DEFINED);



90    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE,



91                          ("CONNECT: invalid mode detected %xH\n", frame->mode));



92



93    /* Everything is fine -> updating the Protocol Layer status */



94    XCP_PRINTF("CONNECT: entering XCP_CONNECTED status\n");



95    xcpStatusSet(XCP_CONNECTED);



96



97    protoErrorCode = XCP_PROTO_SUCCESS;



98    *outputPacketSize = XCP_CONNECT_RES_PACKET_SIZE_IN_BYTES;



99



100    return protoErrorCode;



101}

103static void connectOutputPacketHandler(XcpProtoErrorCode inputCode, void *packet, size_t packetSize)



104{



105    if (inputCode == XCP_PROTO_SUCCESS) {



106        /* Fill connect response */



107        XcpConnectResPacketFrame *frame = (XcpConnectResPacketFrame *)packet;



108        XCP_MEMSET(frame, 0, sizeof(*frame));



109



110        frame->PID = XCP_PID_RES;



111



112        /* Update resource field */



113        frame->resource = xcpResourceValue;



114



115        /* Update commModeBasic field */



116        frame->commModeBasic = xcpCommModeBaseValue;



117



118        /* Update Max CTO and DTO fields */



119        frame->maxCtoSize = (uint8_T)xcpTransportMaxCtoSize();



120        frame->maxDtoSize = (uint16_T)xcpTransportMaxDtoSize();



121



122        /* Update Protocol and Transport Version fields */



123        frame->xcpProtocolVersion  = XCP_MAJOR_NUMBER(XCP_PROTOCOL_LAYER_VERSION);



124        frame->xcpTransportVersion = XCP_MAJOR_NUMBER(XCP_TRANSPORT_LAYER_VERSION);



125



126        XCP_PRINTF("* Resource:          %xH\n", frame->resource);



127        XCP_PRINTF("* Comm Mode Basic:   %xH\n", frame->commModeBasic);



128        XCP_PRINTF("* Max CTO size:      %d\n",  frame->maxCtoSize);



129        XCP_PRINTF("* Max DTO size:      %d\n",  frame->maxDtoSize);



130        XCP_PRINTF("* Protocol Version:  %d\n",  frame->xcpProtocolVersion);



131        XCP_PRINTF("* Transport Version: %d\n",  frame->xcpTransportVersion);



132    }



133    else {



134        genericOutputPacketHandler(inputCode, packet, packetSize);



135    }



136}

141static XcpProtoErrorCode disconnectInputPacketHandler(void   *msgBuffer,



142                                                      size_t  xcpPacketOffset,



143                                                      size_t *outputPacketSize)



144{



145    boolean_T ok = true;



146



147    XCP_UNUSED_PARAM(msgBuffer);



148    XCP_UNUSED_PARAM(xcpPacketOffset);



149



150    /* If DAQ List support is enabled, reset the status of dynamic DAQ list data structures */



151    ok = xcpResetDaqListStatus();



152



153    XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_BUSY,



154        ("DISCONNECT: DAQ lists reset cannot be carried out, as an event was processed\n"));



155



156    /* Everything is fine -> proceed with the status update */



157    XCP_PRINTF("DISCONNECT: entering XCP_DISCONNECTED status\n");



158    xcpStatusSet(XCP_DISCONNECTED);



159



160    *outputPacketSize = XCP_GENERIC_RES_PACKET_SIZE_IN_BYTES;



161



162    return XCP_PROTO_SUCCESS;



163}

182static void getStatusOutputPacketHandler(XcpProtoErrorCode inputCode, void *packet, size_t packetSize)



183{



184    XcpGetStatusResPacketFrame *frame = (XcpGetStatusResPacketFrame *)packet;



185



186    XCP_UNUSED_PARAM(inputCode);



187    XCP_UNUSED_PARAM(packetSize);



188



189    /* Fill Get Status response */



190    XCP_MEMSET(frame, 0, sizeof(*frame));



191



192    frame->PID                      = XCP_PID_RES;



193    frame->sessionStatus            = xcpSessionStatusGet();



194    frame->resourceProtectionStatus = xcpResourceProtectionStatusGet();



195    frame->sessionConfigurationId   = xcpSessionConfigurationIdGet();



196



197    XCP_PRINTF("* Session Status:              %xH\n", frame->sessionStatus);



198    XCP_PRINTF("* Resource Protection Status:  %xH\n", frame->resourceProtectionStatus);



199    XCP_PRINTF("* Session Config Id:           %d\n", frame->sessionConfigurationId);



200}

213



214    XCP_PRINTF("SYNCH\n");



215



216    /* Re-synchronizing the transport layer, by flushing the TX packets currently

253    static XcpProtoErrorCode uploadInputPacketHandler(void   *msgBuffer,



254        size_t  xcpPacketOffset,



255        size_t *outputPacketSize) {



256



257        XcpProtoErrorCode protoErrorCode = XCP_PROTO_SUCCESS;



258        uint8_T *packet = (uint8_T *)msgBuffer + xcpPacketOffset;



259        XcpUploadCmdPacketFrame *frame = (XcpUploadCmdPacketFrame *)packet;



260        boolean_T ok = false;



261



262        /* Check if the number of data elements is valid */



263        



264        ok = isValidUploadSize((uint8_T)frame->size);



265        XCP_INPUT_PKT_ERROR_IF(!ok, XCP_PROTO_OUT_OF_RANGE,



266            ("UPLOAD: invalid number of data elements (%d)\n", (uint8_T)frame->size));



267        



268        xcpUploadSize = (uint8_T)frame->size;



269        *outputPacketSize = XCP_UPLOAD_RES_PACKET_SIZE_IN_BYTES + XCP_IN_BYTES(xcpUploadSize);



270



271        return protoErrorCode;



272    }

275    static void uploadOutputPacketHandler(XcpProtoErrorCode inputCode, void *packet, size_t packetSize) {



276        



277        if (inputCode == XCP_PROTO_SUCCESS) {



278



279            /* Fill Upload response */



280            uint8_T *xcpMTARawPointer = xcpStandardGetAddressFromMta();



281            XcpUploadResPacketFrame *frame = (XcpUploadResPacketFrame *)packet;



282            XCP_MEMSET(frame, 0, sizeof(*frame));



283            frame->PID = XCP_PID_RES;



284



285            if ((xcpMTARawPointer != NULL) &&



286                (packetSize == (XCP_UPLOAD_RES_PACKET_SIZE_IN_BYTES + XCP_IN_BYTES(xcpUploadSize)))



287                ) {



288                uint8_T *pktData = (uint8_T *)packet + XCP_IN_AG(XCP_UPLOAD_RES_PACKET_SIZE_IN_BYTES);



289



290                XCP_PRINTF("UPLOAD: reading at address %p\n", xcpMTARawPointer);



291



292                /* Copy the memory content to the packet data area */



293                XCP_MEMCPY(pktData, xcpMTARawPointer, xcpUploadSize);



294



295                /* Post-increment MTA by the frame size in address granularity units */



296                xcpStandardIncrementMta(xcpUploadSize);



297                xcpUploadSize = 0;



298            }



299            else {



300                /* This is a software error and it should never happen */



301                XCP_PRINTF("UPLOAD: invalid data detected\n");



302            }



303        }



304        else {



305            genericOutputPacketHandler(inputCode, packet, packetSize);



306        }



307    }



308#endif  /* XCP_SET_MTA_ENABLE == 1  */

411void xcpStandardInit(void)



412{



413    /* Initialize the packet lookup function to support only basic



414       standard commands */



415    xcpStandardSetPacketLookup(getPacket);



416



417    /* Initialize support for the extended list of Standard commands



418       @note this may override the default Packet lookup function



419             by adding support for more (optional) commands */



420    xcpStandardExtendedInit();



421}

446uint8_T* xcpStandardGetAddressFromMta(void) {



447    return XCP_ADDRESS_GET(xcpMta.addressExtension, xcpMta.address);

451boolean_T xcpStandardSetMta(uint32_T address, uint8_T addressExtension) {



452    if ((addressExtension == 0) && (address == 0)) {



453        return false;



454    } else {



455        xcpMta.address = address;



456        xcpMta.addressExtension = addressExtension;



457        return true;



458    }



459}
