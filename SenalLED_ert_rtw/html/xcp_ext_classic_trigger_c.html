60/** The function is responsible for sampling the signal trigger value



61    and setting the xcpSignalTriggerFired boolean to true if



62    the trigger signal has the expected transition across the xcpClassicTriggerLevel.



63    The boolean will always be restored back to false by the base rate



64    when the event is acknowledged and the state transition completed.



65    The function is also setting xcpTriggerEnabled to true if no Delay is required. */



66static void sampleSignalTrigger(void)



67{



68    extmodeClassicTriggerSignal_T *signal = (extmodeClassicTriggerSignal_T *)



69                                            XCP_ADDRESS_GET(xcpClassicTriggerSignalAddressExtension,



70                                                            xcpClassicTriggerSignalAddress);



71    if ((signal != NULL) &&



72       ((xcpClassicTriggerSignalAddressExtension != 0) ||



73        (xcpClassicTriggerSignalAddress != 0))) {



74        if ((xcpClassicTriggerStatus ==	XCP_EXTMODE_TRIGGER_ARMED) &&



75            !xcpSignalTriggerFired) {



76            /* Read the new signal value */



77            extmodeClassicTriggerSignal_T signalValue = *signal;



78



79            if (xcpOldSignalValueAvailable) {



80                /* If the trigger is armed and not already fired,



81                   check for the expected transition */



82                boolean_T checkForRising =  (xcpClassicTriggerDirection == XCP_EXTMODE_TRIGGER_RISING) ||



83                                            (xcpClassicTriggerDirection == XCP_EXTMODE_TRIGGER_RISING_OR_FALLING);



84



85                boolean_T checkForFalling = (xcpClassicTriggerDirection == XCP_EXTMODE_TRIGGER_FALLING) ||



86                                            (xcpClassicTriggerDirection == XCP_EXTMODE_TRIGGER_RISING_OR_FALLING);



87



88                if (checkForRising &&



89                   (((signalValue >= xcpClassicTriggerLevel) && (xcpOldSignalValue <  xcpClassicTriggerLevel)) ||



90                    ((signalValue >  xcpClassicTriggerLevel) && (xcpOldSignalValue == xcpClassicTriggerLevel)))) {



91



92                    if (xcpClassicTriggerDelay == 0) {



93                        /* fire the event immediately if needed no Delay is required */



94                        xcpTriggerEnabled = true;



95                    }



96                    xcpSignalTriggerFired   = true; /* the transition from ARMED to FIRING is updated in base rate */



97                }



98                if (checkForFalling &&



99                   (((signalValue < xcpClassicTriggerLevel)  && (xcpOldSignalValue >= xcpClassicTriggerLevel)) ||



100                    ((signalValue == xcpClassicTriggerLevel) && (xcpOldSignalValue >  xcpClassicTriggerLevel)))) {



101



102                    if (xcpClassicTriggerDelay == 0) {



103                        /* fire the event immediately if needed no Delay is required */



104                        xcpTriggerEnabled = true;



105                    }



106                    xcpSignalTriggerFired   = true; /* the transition from ARMED to FIRING is updated in base rate */



107                }



108            }



109



110            /* Update the old signal value. Note: if we just fired, the sampling will start



111               next time the trigger is armed and therefore we reset xcpOldSignalValueAvailable */



112            xcpOldSignalValueAvailable = !xcpSignalTriggerFired;



113            xcpOldSignalValue = signalValue;



114        }



115    }



116}

119/** The function is executed in the base rate and it is responsible



120    for the handling of the triggering status transitions */



121static void updateTriggerStatus(void)



122{



123    /* Process user Arm/Cancel requests common to all states



124       and update the triggerFired variable (depending on the



125       active triggering source) */



126    boolean_T triggerFired = false;



127



128    if (xcpClassicTriggerStatus == XCP_EXTMODE_TRIGGER_UNARMED) {



129        /* This case deserves a separate handling



130           because we would like the transition to the ARMED status



131           to occur in the same cycle.



132           Since we are already UNARMED, any request to cancel 



133           the trigger can be ignored */



134        xcpClassicTriggerCancelRequest = false;



135        xcpSignalTriggerFired = false;



136



137        if (xcpClassicTriggerArmRequest) {



138            /* Process arm request */



139            xcpClassicTriggerStatus = XCP_EXTMODE_TRIGGER_ARMED;



140            xcpClassicTriggerArmRequest = false;



141



142            /* If we are using MANUAL trigger, we want to start 



143               firing the trigger event straightaway, e.g.



144               when the user selects ExtModeArmWhenConnect */



145            if (xcpClassicTriggerSource == XCP_EXTMODE_TRIGGER_MANUAL) {



146                triggerFired = true;



147            }



148        }



149    } else {



150        /* We are already armed, ignore any request to arm the trigger */



151        xcpClassicTriggerArmRequest = false;



152



153        if (xcpClassicTriggerCancelRequest) {



154            /* No matter what the Status is, a cancel request should



155               always bring the status to UNARMED.



156               We implement the transition here, to avoid code



157               duplication in the states. */



158            xcpTriggerEnabled = false;



159            xcpClassicTriggerStatus = XCP_EXTMODE_TRIGGER_UNARMED;



160            xcpClassicTriggerCancelRequest = false;



161        } else {



162            if (xcpClassicTriggerSource == XCP_EXTMODE_TRIGGER_MANUAL) {



163                /* when manual triggering is selected, we assume that



164                   the trigger event has always been fired when armed */



165                triggerFired = true;



166            } else {



167                /* when signal triggering is selected, we check the status



168                   of the xcpSignalTriggerFired (updated by the



169                   sampleSignalTrigger() logic */



170                triggerFired = xcpSignalTriggerFired;



171                xcpSignalTriggerFired = false;



172            }



173        }



174    }



175



176    /* Handle the remaining state transitions */



177    switch (xcpClassicTriggerStatus) {



178    case XCP_EXTMODE_TRIGGER_UNARMED:



179        if (xcpClassicTriggerArmRequest) {



180            /* Process arm request */



181            xcpClassicTriggerStatus = XCP_EXTMODE_TRIGGER_ARMED;



182            xcpClassicTriggerArmRequest = false;



183        }



184        break;



185



186    case XCP_EXTMODE_TRIGGER_ARMED:



187        if (triggerFired) {



188            if ((xcpClassicTriggerDelay > 0) &&



189                (xcpClassicTriggerSource == XCP_EXTMODE_TRIGGER_SIGNAL)) {



190                /* We need to postpone the enabling of the trigger



191                   by xcpClassicTriggerDelay base periods */



192                xcpTriggerCount = (uint32_T) xcpClassicTriggerDelay;



193                xcpClassicTriggerStatus = XCP_EXTMODE_TRIGGER_DELAYED;



194            } else {



195                /* Negative xcpClassicTriggerDelay are ignored.



196                   The parameter is also ignored if the xcpClassicTriggerSource



197                   is XCP_EXTMODE_TRIGGER_MANUAL



198                   If xcpClassicTriggerDelay is 0 we enable the trigger straight away */



199                if (xcpClassicTriggerDuration > 0) {



200                    xcpTriggerEnabled = true;



201                    xcpTriggerCount = xcpClassicTriggerDuration;



202                    xcpClassicTriggerStatus = XCP_EXTMODE_TRIGGER_FIRING;



203                } else {



204                    /* If duration is 0, we would never be able to send any data.



205                       This situation should be prevented on the host,



206                       we just handle it gracefully here, in case it happens */



207                    XCP_PRINTF("Invalid duration value (0) detected.\n");



208                    xcpTriggerEnabled = false;



209                    xcpClassicTriggerStatus = XCP_EXTMODE_TRIGGER_UNARMED;



210                }



211            }



212        }



213        break;



214



215    case XCP_EXTMODE_TRIGGER_FIRING:



216        /* Update the base rate counter */



217        if (xcpTriggerCount > 0) {



218            xcpTriggerCount--;



219        }



220



221        if (xcpTriggerCount == 0) {



222            /* Duration has expired */



223            if (xcpClassicTriggerMode == XCP_EXTMODE_TRIGGER_ONESHOT) {



224                /* In the one-shot scenario we just disable the trigger */



225                xcpTriggerEnabled = false;



226                xcpClassicTriggerStatus  = XCP_EXTMODE_TRIGGER_UNARMED;



227            } else {



228                /* In normal mode, we might have to handle the HoldOff and



229                   pause the triggering  */



230                if (xcpClassicTriggerHoldOff > 0) {



231                    xcpTriggerEnabled = false;



232                    xcpTriggerCount = xcpClassicTriggerHoldOff;



233                    xcpClassicTriggerStatus = XCP_EXTMODE_TRIGGER_HOLDING_OFF;



234                } else {



235                    if (triggerFired) {



236                        /* If the trigger has been fired, we start another duration



237                           straight away */



238                        xcpTriggerCount = xcpClassicTriggerDuration;



239                        xcpClassicTriggerStatus = XCP_EXTMODE_TRIGGER_FIRING;



240                    } else {



241                        /* We wait for the next trigger to be fired */



242                        xcpTriggerEnabled = false;



243                        xcpClassicTriggerStatus = XCP_EXTMODE_TRIGGER_ARMED;



244                    }



245                }



246            }



247        }



248        break;



249



250    case XCP_EXTMODE_TRIGGER_DELAYED:



251        /* Update the base rate counter */



252        if (xcpTriggerCount > 0) {



253            xcpTriggerCount--;



254        }



255



256        if (xcpTriggerCount == 0) {



257            /* Delay is over, we can enable the trigger now */



258            if (xcpClassicTriggerDuration > 0) {



259                xcpTriggerEnabled = true;



260                xcpTriggerCount = xcpClassicTriggerDuration;



261                xcpClassicTriggerStatus = XCP_EXTMODE_TRIGGER_FIRING;



262            } else {



263                /* If duration is 0, we would never be able to send any data.



264                   This situation should be prevented on the host,



265                   we just handle it gracefully here, in case it happens */



266                XCP_PRINTF("Invalid duration value (0) detected.\n");



267                xcpTriggerEnabled = false;



268                xcpClassicTriggerStatus = XCP_EXTMODE_TRIGGER_UNARMED;



269            }



270        }



271        break;



272



273    case XCP_EXTMODE_TRIGGER_HOLDING_OFF:



274        /* Update the base rate counter */



275        if (xcpTriggerCount > 0) {



276            xcpTriggerCount--;



277        }



278



279        if (xcpTriggerCount == 0) {



280            /* HoldOff time has expired */



281            if (triggerFired) {



282                /* If the trigger has been fired, we start another duration



283                   straight away */



284                if (xcpClassicTriggerDuration > 0) {



285                    xcpTriggerEnabled = true;



286                    xcpTriggerCount = xcpClassicTriggerDuration;



287                    xcpClassicTriggerStatus = XCP_EXTMODE_TRIGGER_FIRING;



288                } else {



289                    /* If duration is 0, we would never be able to send any data.



290                       This situation should be prevented on the host,



291                       we just handle it gracefully here, in case it happens */



292                    XCP_PRINTF("Invalid duration value (0) detected.\n");



293                    xcpTriggerEnabled = false;



294                    xcpClassicTriggerStatus = XCP_EXTMODE_TRIGGER_UNARMED;



295                }



296            } else {



297                /* We wait for the next trigger to be fired */



298                xcpClassicTriggerStatus = XCP_EXTMODE_TRIGGER_ARMED;



299            }



300        }



301        break;



302



303    default:



304        XCP_PRINTF("Invalid Trigger Status %d detected\n", xcpClassicTriggerStatus);



305        xcpTriggerEnabled = false;



306        xcpSignalTriggerFired = false;



307        xcpClassicTriggerCancelRequest = false;



308        xcpClassicTriggerStatus = XCP_EXTMODE_TRIGGER_UNARMED;



309    }

331/** The function returns true if the external mode event must be triggered */



332boolean_T xcpExtModeClassicTriggerEnabled(extmodeEventId_T eventId)



333{



334    if (eventId == EXTMODE_TRIG_BASE_RATE_EVENT_ID) {



335        /* Whenever a change of the xcpClassicTriggerSource is detected



336           the xcpClassicTriggerStatus is reset */



337        boolean_T triggerSrcChangeDetected = (xcpClassicTriggerSource != xcpOldClassicTriggerSource);



338        xcpOldClassicTriggerSource = xcpClassicTriggerSource;



339



340        if (triggerSrcChangeDetected) {



341            xcpTriggerEnabled     = false;



342            xcpSignalTriggerFired = false;



343            xcpTriggerCount       = 0;



344



345            xcpClassicTriggerStatus  = XCP_EXTMODE_TRIGGER_UNARMED;



346        }



347



348        /* If XCP_EXTMODE_TRIGGER_SIGNAL is selected we want to start



349           processing the Arm request (or the Hold-off expiration) 



350           straightaway and carry out the state transition before 



351           sampling for the first time */



352        if ((xcpClassicTriggerSource == XCP_EXTMODE_TRIGGER_SIGNAL)  &&



353            (((xcpClassicTriggerStatus == XCP_EXTMODE_TRIGGER_UNARMED) &&



354               xcpClassicTriggerArmRequest) ||



355             ((xcpClassicTriggerStatus == XCP_EXTMODE_TRIGGER_HOLDING_OFF) &&



356              (xcpTriggerCount == 1)))) {



357               xcpTriggerCount = 0;



358               xcpClassicTriggerArmRequest = false;



359               xcpClassicTriggerStatus = XCP_EXTMODE_TRIGGER_ARMED;



360        }



361    }



362



363    if ((xcpClassicTriggerSource == XCP_EXTMODE_TRIGGER_SIGNAL) &&



364        (eventId == xcpClassicTriggerEventId))  {



365        /* Since we are running at the correct eventID, sample the



366           trigger signal value to determine if the trigger signal



367           needs to be enabled.



368           Note: this will update the boolean straight away and request



369           the state transition (updated at the next base rate) */



370        sampleSignalTrigger();



371    }



372



373    if (eventId == EXTMODE_TRIG_BASE_RATE_EVENT_ID) {



374        /* update trigger status to fulfill user requests */



375        updateTriggerStatus();



376    }



377



378    return xcpTriggerEnabled;



379}

382/** Reset External Mode classic triggering logic, by restoring the default status */



383XcpErrorCode xcpExtModeClassicTriggerReset(void)



384{



385    xcpOldSignalValue = 0;



386    xcpOldSignalValueAvailable = false;



387



388    xcpClassicTriggerStatus  = XCP_EXTMODE_TRIGGER_UNARMED;



389    xcpOldClassicTriggerSource = XCP_EXTMODE_TRIGGER_MANUAL;



390    xcpClassicTriggerEventId = 0;



391    xcpTriggerEnabled = false;



392    xcpSignalTriggerFired = false;



393    xcpTriggerCount = 0;



394



395    xcpClassicTriggerSignalAddress = 0;



396    xcpClassicTriggerSignalAddressExtension = 0;



397



398    xcpClassicTriggerLevel    = 0;



399    xcpClassicTriggerDuration = DEFAULT_XCP_CLASSIC_TRIGGER_DURATION;



400    xcpClassicTriggerHoldOff  = 0;



401    xcpClassicTriggerDelay    = 0;



402



403    xcpClassicTriggerDirection = XCP_EXTMODE_TRIGGER_RISING;



404



405#if defined(EXTMODE_TRIG_ARMED_ON_START) && EXTMODE_TRIG_ARMED_ON_START == 0



406    xcpClassicTriggerArmRequest = false;



407#else



408/* Classic Trigger is armed by default */



409    xcpClassicTriggerArmRequest = true;



410#endif



411



412    xcpClassicTriggerCancelRequest = false;



413



414    xcpClassicTriggerSource = XCP_EXTMODE_TRIGGER_MANUAL;



415    xcpClassicTriggerMode = XCP_EXTMODE_TRIGGER_NORMAL;



416



417    return XCP_SUCCESS;
